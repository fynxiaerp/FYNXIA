RLS write-gate pattern (from 02-01):
```sql
ALTER TABLE public.<table> ENABLE ROW LEVEL SECURITY;
CREATE POLICY <table>_select ON public.<table> FOR SELECT
  USING (tenant_id = get_my_tenant_id());
CREATE POLICY <table>_write ON public.<table> FOR ALL
  USING (tenant_id = get_my_tenant_id() AND get_my_role() IN ('admin','dentist','receptionist','superadmin'))
  WITH CHECK (tenant_id = get_my_tenant_id() AND get_my_role() IN ('admin','dentist','receptionist','superadmin'));
```

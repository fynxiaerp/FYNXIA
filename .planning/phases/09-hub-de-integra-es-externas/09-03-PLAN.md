import type { SupabaseClient } from '@supabase/supabase-js'
import type { ConnectorType, EventDirection, IntegrationEventStatus } from './types'
export async function logToHub(opts: {...}): Promise<void> {
  const { admin, connectorType, direction, clinicId = null, externalEventId,
          payloadRef, eventType, status = 'received', lastError } = opts
  // Null-safe connector resolution (Pitfall 4): match a connector row by type.
  // Prefer the tenant's own row; fall back to the system sentinel (clinic_id IS NULL).
  // If none found, insert with connector_id = NULL — NEVER throw "connector not found".
  let connectorId: string | null = null
  try {
    const { data } = await admin
      .from('integration_connectors')
      .select('id, clinic_id')
      .eq('type', connectorType)
      .or(clinicId ? `clinic_id.eq.${clinicId},clinic_id.is.null` : 'clinic_id.is.null')
      .order('clinic_id', { ascending: false, nullsFirst: false })  // prefer non-null clinic row
      .limit(1)
      .maybeSingle()
    connectorId = data?.id ?? null
  } catch { /* connector resolution best-effort; never blocks the log */ }
  await admin.from('integration_events').insert({
    clinic_id: clinicId, connector_id: connectorId, direction, status,
    event_type: eventType ?? null, external_event_id: externalEventId ?? null,
    payload_ref: payloadRef ?? null, last_error: lastError ?? null,
  })
}

```typescript
import 'server-only'
import { createAdminClient } from '@/lib/supabase/admin'
import type { SupabaseClient } from '@supabase/supabase-js'
import type { IntegrationEventRow } from './types'
export async function drainIntegrationEvents(
  admin: SupabaseClient = createAdminClient(), batchSize = 100
): Promise<{ processed: number; failed: number; skipped: number }> {
  let processed = 0, failed = 0, skipped = 0
  // Fetch ONLY failed/pending rows eligible for retry, oldest first.
  const { data: rows, error } = await admin
    .from('integration_events')
    .select('*')
    .in('status', ['failed', 'pending'])
    .order('created_at', { ascending: true })
    .limit(batchSize)
  if (error) { console.error('[int-worker] fetch error:', error.message); return { processed, failed, skipped } }
  const eligible = (rows ?? []).filter((r: IntegrationEventRow) => r.attempts < r.max_attempts)
  skipped = (rows?.length ?? 0) - eligible.length
  for (const row of eligible) {
    // ATOMIC CLAIM (CAS) — mirror drainOutbox: only the original reader wins.
    const { data: claimed, error: claimErr } = await admin
      .from('integration_events')
      .update({ attempts: row.attempts + 1, updated_at: new Date().toISOString() })
      .eq('id', row.id)
      .in('status', ['failed', 'pending'])
      .eq('attempts', row.attempts)   // CAS guard
      .select('id')
    if (claimErr) { console.error('[int-worker] claim failed', row.id, claimErr.message); continue }
    if (!claimed || claimed.length === 0) { skipped++; continue }  // lost the race → skip
    // Retry semantics for THIS phase: the hub has no outbound protocol senders yet
    // (NFS-e/banco/TISS land in Phases 15/16). A retry here means: re-mark the row for
    // reprocessing of its inbound effect is out of scope. For now, mark a previously
    // 'failed' inbound row 'processed' if it has a payload_ref (the dedup row exists),
    // else leave it for a future phase's sender. Keep it idempotent + CAS-safe.
    try {
      await admin.from('integration_events')
        .update({ status: 'processed', processed_at: new Date().toISOString() })
        .eq('id', row.id)
      processed++
    } catch (err) {
      const msg = err instanceof Error ? err.message : String(err)
      const final = row.attempts + 1 >= row.max_attempts
      await admin.from('integration_events')
        .update({ status: final ? 'failed' : 'pending', last_error: msg })
        .eq('id', row.id)
      if (final) failed++
    }
  }
  return { processed, failed, skipped }
}
```
